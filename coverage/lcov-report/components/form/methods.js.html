function decimalPlaces(num) {
    var match = ("" + num).match(/(?:\.(\d+))?$/);
    if (!match) {
        return 0;
    }
 
    // Number of digits right of decimal point.
    return match[1] ? match[1].length : 0;
}

export const addMethod = function(name, method, message, className) {
